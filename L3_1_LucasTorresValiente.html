function[y] = fexponencial(x)
for k = 1:length(x)
y(k) = 0;
error_abs(k) = abs(y(k)-(exp(x(k))));
sumands(k) = 0;
i = zeros(size(x));
while(i(k) <= 100 & error_abs(k) >= 0.5e-12)
y(k) = y(k) + (x(k).^i(k))./factorial(i(k));
error_abs(k) = abs(y(k)-(exp(x(k)))); %actualització de l'error absolut
i(k) = i(k) + 1; %actualització de l'index per iterar
sumands(k) = i(k);
end
end
sumands
end
%Donat que es demanda que la funció només tingui un argument de sortida, per a esciure

function[y,sumands] = fexponencial_amb_iteracions(x)
for k = 1:length(x)
y(k) = 0;
error_abs(k) = abs(y(k)-(exp(x(k))));
sumands(k) = 0;
i = zeros(size(x));
while(i(k) <= 100 & error_abs(k) >= 0.5e-12)
y(k) = y(k) + (x(k).^i(k))./factorial(i(k));
error_abs(k) = abs(y(k)-(exp(x(k)))); %actualització de l'error absolut
i(k) = i(k) + 1; %actualització de l'index per iterar
sumands(k) = i(k);
end
end
